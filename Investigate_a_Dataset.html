function to plot top then counts of a column¶

def barplot_counts_top_10(column, data=df):
    values_count = data[column].value_counts()
    values_count = values_count[:10, ]
    plt.figure(figsize=(10,8))
    sns.barplot(values_count.index, values_count.values, color='blue')
